# Creating function to make code modular
def answer1():
    df_first1000 = df.head(1000)
    ans1 = df_first1000['channeltype'].value_counts(normalize=False)
    result = pd.DataFrame({
        'Channel Type': ans1.index,
        'Count': ans1.values
    })
    return result
